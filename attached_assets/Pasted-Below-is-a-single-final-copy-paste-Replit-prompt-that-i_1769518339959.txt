```pseudo
type Branding = { displayName, subtitle, logoUrl, themeJson, source }

function getDefaultBranding():
  return { displayName:"GeoHealthSync", subtitle:"Municipal Health Information System", logoUrl:null, themeJson:null, source:"DEFAULT" }

function safeTrim(s):
  if s == null return null
  t = trim(s)
  if t == "" return null
  return t

function coalesceNonEmpty(a,b,c):
  if safeTrim(a)!=null return safeTrim(a)
  if safeTrim(b)!=null return safeTrim(b)
  if safeTrim(c)!=null return safeTrim(c)
  return null

async function resolveBranding({scope, barangayId, municipalityId}):
  defaultBranding = getDefaultBranding()

  muni = db.municipality_settings.findUnique(where municipalityId)
  muniDisplayName = coalesceNonEmpty(muni?.municipalityName, null, defaultBranding.displayName)
  muniSubtitle    = coalesceNonEmpty(muni?.subtitle, null, defaultBranding.subtitle)
  muniLogoUrl     = safeTrim(muni?.logoUrl)
  muniThemeJson   = muni?.themeJson ?? null

  if scope == "municipality":
    logo = coalesceNonEmpty(muniLogoUrl, defaultBranding.logoUrl, null)
    return { displayName:muniDisplayName, subtitle:muniSubtitle, logoUrl:logo, themeJson:muniThemeJson, source:(muni!=null?"MUNICIPALITY":"DEFAULT") }

  // scope == "barangay"
  assert(barangayId != null)
  brgy = db.barangays.findUnique(where id=barangayId)
  assert(brgy != null)

  brgySettings = db.barangay_settings.findUnique(where barangayId)
  brgyDisplayName = coalesceNonEmpty(brgySettings?.barangayNameOverride, brgy?.name, "Barangay (Unnamed)")
  brgySubtitle    = coalesceNonEmpty(brgySettings?.subtitle, muniSubtitle, defaultBranding.subtitle)
  brgyLogoUrl     = safeTrim(brgySettings?.logoUrl)

  resolvedLogoUrl  = coalesceNonEmpty(brgyLogoUrl, muniLogoUrl, defaultBranding.logoUrl)
  resolvedThemeJson = brgySettings?.themeJson ?? muniThemeJson ?? defaultBranding.themeJson

  source = (brgySettings!=null || brgyLogoUrl!=null) ? "BARANGAY"
         : (muni!=null || muniLogoUrl!=null) ? "MUNICIPALITY"
         : "DEFAULT"

  return { displayName:brgyDisplayName, subtitle:brgySubtitle, logoUrl:resolvedLogoUrl, themeJson:resolvedThemeJson, source:source }

function drawM1HeaderFirstPage(pdf, page, {branding, reportTitle, periodLabel, scopeLabel, barangayLabel}):
  // LOGO ONLY ON PAGE 1
  if branding.logoUrl != null:
    pdf.drawImage(branding.logoUrl, x=24, y=24, w=48, h=48)

  centerX = page.width / 2
  pdf.setFont("Helvetica", 11, bold=true)
  pdf.drawText("Field Health Services Information System (FHSIS)", x=centerX, y=24, align="center")
  pdf.drawText(branding.displayName, x=centerX, y=38, align="center")

  if branding.subtitle != null:
    pdf.setFont("Helvetica", 10, bold=false)
    pdf.drawText(branding.subtitle, x=centerX, y=52, align="center")

  pdf.setFont("Helvetica", 12, bold=true)
  pdf.drawText(reportTitle, x=centerX, y=68, align="center")

  pdf.setFont("Helvetica", 10, bold=false)
  pdf.drawText(periodLabel, x=centerX, y=84, align="center")
  pdf.drawText(scopeLabel, x=centerX, y=96, align="center")

  if barangayLabel != null:
    pdf.drawText(barangayLabel, x=centerX, y=108, align="center")

  pdf.drawLine(x1=24, y1=120, x2=page.width-24, y2=120)

function drawCompactHeaderNoLogo(pdf, page, {title, period, scope}):
  // PAGES 2–3: NO LOGO
  pdf.setFont("Helvetica", 10, bold=true)
  pdf.drawText(title + " | " + period + " | " + scope, x=24, y=24, align="left")
  pdf.drawLine(x1=24, y1=32, x2=page.width-24, y2=32)

async function generateM1Pdf(params):
  authorizeUserForReport(params.viewerUserId, params.scope, params.barangayId)

  branding = await resolveBranding({scope:params.scope, barangayId:params.barangayId, municipalityId:params.municipalityId})
  reportData = await getM1ReportData(params) // computed or imported

  pdf = createPdfDocument(A4_portrait)
  for pageIndex in [0,1,2]:
    page = pdf.addPage()

    if pageIndex == 0:
      drawM1HeaderFirstPage(pdf, page, {
        branding,
        reportTitle:"FHSIS Monthly Report (M1 / M1Brgy)",
        periodLabel:"Reporting Period: " + monthName(params.month) + " " + params.year,
        scopeLabel:(params.scope=="barangay" ? "Scope: Barangay Report" : "Scope: Municipality Consolidated"),
        barangayLabel:(params.scope=="barangay" ? "Barangay: " + getBarangayName(params.barangayId) : null)
      })
      renderM1BodyPage1(pdf,page,reportData)
    else:
      drawCompactHeaderNoLogo(pdf,page,{title:"M1 / M1Brgy", period:monthName(params.month)+" "+params.year, scope:(params.scope=="barangay"?getBarangayName(params.barangayId):"Municipality Consolidated")})
      if pageIndex==1 renderM1BodyPage2(pdf,page,reportData)
      if pageIndex==2 renderM1BodyPage3(pdf,page,reportData)

  return pdfBuffer(pdf)
```

model User {

name          String
email         String   @unique
passwordHash  String
role          Role
status        UserStatus @default(ACTIVE)
createdAt     DateTime @default(now())
updatedAt     DateTime @updatedAt
barangayLinks UserBarangay[]
auditLogs     AuditLog[]
consults      Consult[] @relation("DoctorConsults")
}

model Barangay {
id        String @id @default(cuid())
name      String
createdAt DateTime @default(now())
users     UserBarangay[]
settings  BarangaySettings?
}

model UserBarangay {
userId    String
barangayId String
user      User @relation(fields:[userId], references:[id])
barangay  Barangay @relation(fields:[barangayId], references:[id])

}

enum Role { SYSTEM_ADMIN MHO SHA TL }
