// ==============================
// TYPES
// ==============================
type Branding = {
  displayName: string              // "Barangay X" or "Municipality of Y"
  subtitle: string | null
  logoUrl: string | null           // may be null if no logo available
  themeJson: object | null         // optional
  source: "BARANGAY" | "MUNICIPALITY" | "DEFAULT"
}

type ResolveBrandingInput = {
  scope: "barangay" | "municipality"
  barangayId: string | null
  municipalityId: string           // single LGU, or from tenant context
}

type ResolveBrandingOutput = Branding


// ==============================
// HELPERS
// ==============================
function getDefaultBranding(): Branding
  return {
    displayName: "GeoHealthSync",
    subtitle: "Municipal Health Information System",
    logoUrl: null,                     // or built-in asset path
    themeJson: null,
    source: "DEFAULT"
  }

function safeTrim(s: string | null): string | null
  if s == null then return null
  let t = trim(s)
  if t == "" then return null
  return t

function coalesceNonEmpty(a, b, c)
  // returns first non-null, non-empty string
  if safeTrim(a) != null then return safeTrim(a)
  if safeTrim(b) != null then return safeTrim(b)
  if safeTrim(c) != null then return safeTrim(c)
  return null

function assert(condition: boolean, message: string)
  if !condition then throw Error(message)


// ==============================
// CORE: BRANDING RESOLUTION
// ==============================
async function resolveBranding(input: ResolveBrandingInput): ResolveBrandingOutput
  assert(input.municipalityId != null, "municipalityId required")
  assert(input.scope in ["barangay", "municipality"], "invalid scope")

  let defaultBranding = getDefaultBranding()

  // Always try municipality settings (used for consolidated + fallback)
  let muni = await db.municipality_settings.findUnique({
    where: { municipalityId: input.municipalityId }
  })

  let muniDisplayName = coalesceNonEmpty(muni?.municipalityName, null, defaultBranding.displayName)
  let muniSubtitle    = coalesceNonEmpty(muni?.subtitle, null, defaultBranding.subtitle)
  let muniLogoUrl     = safeTrim(muni?.logoUrl)
  let muniThemeJson   = muni?.themeJson ?? null

  if input.scope == "municipality"
    // Municipality consolidated report
    let logo = coalesceNonEmpty(muniLogoUrl, defaultBranding.logoUrl, null)
    return {
      displayName: muniDisplayName,
      subtitle: muniSubtitle,
      logoUrl: logo,
      themeJson: muniThemeJson,
      source: (logo != null or muni != null) ? "MUNICIPALITY" : "DEFAULT"
    }

  // scope == "barangay"
  assert(input.barangayId != null, "barangayId required for barangay scope")

  // Load barangay + barangay settings
  let brgy = await db.barangays.findUnique({ where: { id: input.barangayId } })
  assert(brgy != null, "barangay not found")

  let brgySettings = await db.barangay_settings.findUnique({
    where: { barangayId: input.barangayId }
  })

  let brgyNameOverride = safeTrim(brgySettings?.barangayNameOverride)
  let brgyDisplayName  = coalesceNonEmpty(brgyNameOverride, brgy?.name, "Barangay (Unnamed)")
  let brgySubtitle     = coalesceNonEmpty(brgySettings?.subtitle, muniSubtitle, defaultBranding.subtitle)
  let brgyLogoUrl      = safeTrim(brgySettings?.logoUrl)

  // Logo fallback order: barangay logo -> municipality logo -> default logo (optional)
  let resolvedLogoUrl  = coalesceNonEmpty(brgyLogoUrl, muniLogoUrl, defaultBranding.logoUrl)

  // Theme fallback order: barangay theme -> municipality theme -> default theme
  let resolvedThemeJson = brgySettings?.themeJson ?? muniThemeJson ?? defaultBranding.themeJson

  let source =
    (brgyLogoUrl != null or brgySettings != null) ? "BARANGAY"
    : (muniLogoUrl != null or muni != null) ? "MUNICIPALITY"
    : "DEFAULT"

  return {
    displayName: brgyDisplayName,
    subtitle: brgySubtitle,
    logoUrl: resolvedLogoUrl,
    themeJson: resolvedThemeJson,
    source: source
  }

// PDF HEADER RENDER (FIRST PAGE ONLY)
// ==============================
// Assumption: using a PDF library where page coordinates are in points or mm.
// This function is called ONLY on pageIndex == 0.
function drawM1HeaderFirstPage(pdf, page, headerInput)
  // headerInput includes branding + report meta
  // headerInput: { branding, reportTitle, periodLabel, scopeLabel, barangayLabel }

  let marginLeft = 24
  let marginTop  = 24

  // Layout spec: Government-style header, readable, not cramped.
  // - Logo: left
  // - Center block: Republic/DOH/LGU name + report title + period/scope

  if headerInput.branding.logoUrl != null
    // Draw logo at top-left; keep modest size to save space
    // Target size: 48x48 (points) or equivalent
    pdf.drawImage(page, headerInput.branding.logoUrl, x=marginLeft, y=marginTop, w=48, h=48)

  // Center header block (aligned center on page)
  let centerX = page.width / 2

  // Use conservative font sizes to match government forms: 10–12pt
  pdf.setFont("Helvetica", 11, bold=true)
  pdf.drawText(page, "Field Health Services Information System (FHSIS)", x=centerX, y=marginTop, align="center")

  pdf.setFont("Helvetica", 11, bold=true)
  pdf.drawText(page, headerInput.branding.displayName, x=centerX, y=marginTop + 14, align="center")

  if headerInput.branding.subtitle != null
    pdf.setFont("Helvetica", 10, bold=false)
    pdf.drawText(page, headerInput.branding.subtitle, x=centerX, y=marginTop + 28, align="center")

  pdf.setFont("Helvetica", 12, bold=true)
  pdf.drawText(page, headerInput.reportTitle, x=centerX, y=marginTop + 44, align="center")
  // e.g. "FHSIS Monthly Report (M1 / M1Brgy)"

  pdf.setFont("Helvetica", 10, bold=false)
  pdf.drawText(page, headerInput.periodLabel, x=centerX, y=marginTop + 60, align="center")
  // e.g. "Reporting Period: June 2025"

  pdf.setFont("Helvetica", 10, bold=false)
  pdf.drawText(page, headerInput.scopeLabel, x=centerX, y=marginTop + 72, align="center")
  // e.g. "Scope: Barangay Report" or "Scope: Municipality Consolidated"

  if headerInput.barangayLabel != null
    pdf.setFont("Helvetica", 10, bold=false)
    pdf.drawText(page, headerInput.barangayLabel, x=centerX, y=marginTop + 84, align="center")
    // e.g. "Barangay: San Isidro"

  // Optional: thin divider line below header
  pdf.drawLine(page, x1=marginLeft, y1=marginTop + 96, x2=page.width - marginLeft, y2=marginTop + 96)

// PDF GENERATION FLOW (3 pages)
// ==============================
async function generateM1Pdf(params)
  // params: { scope, barangayId?, month, year, municipalityId, viewerUserId }
  // 1) authorize per role + scope
  // 2) compute or load m1 data rows
  // 3) resolve branding
  // 4) render 3-page PDF
  // 5) only page 1 gets logo/header block; pages 2-3 get compact title line

  authorizeUserForReport(params.viewerUserId, params.scope, params.barangayId)

  let branding = await resolveBranding({
    scope: params.scope,
    barangayId: params.barangayId ?? null,
    municipalityId: params.municipalityId
  })

  let reportData = await getM1ReportData(params) // computed or imported

  let pdf = pdfLib.createDocument()
  for pageIndex in [0,1,2]
    let page = pdf.addPage(size="A4", orientation="portrait")

    if pageIndex == 0
      drawM1HeaderFirstPage(pdf, page, {
        branding: branding,
        reportTitle: "FHSIS Monthly Report (M1 / M1Brgy)",
        periodLabel: "Reporting Period: " + monthName(params.month) + " " + params.year,
        scopeLabel: (params.scope == "barangay") ? "Scope: Barangay Report" : "Scope: Municipality Consolidated",
        barangayLabel: (params.scope == "barangay") ? ("Barangay: " + getBarangayName(params.barangayId)) : null
      })
      renderM1BodyPage1(pdf, page, reportData)
    else
      drawCompactHeaderNoLogo(pdf, page, {
        title: "M1 / M1Brgy",
        period: monthName(params.month) + " " + params.year,
        scope: (params.scope == "barangay") ? getBarangayName(params.barangayId) : "Municipality Consolidated"
      })
      if pageIndex == 1 then renderM1BodyPage2(pdf, page, reportData)
      if pageIndex == 2 then renderM1BodyPage3(pdf, page, reportData)

  return pdf.saveToBuffer()
